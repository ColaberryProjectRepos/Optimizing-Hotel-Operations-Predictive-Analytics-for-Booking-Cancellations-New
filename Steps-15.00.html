  1. In Power BI Desktop, click on Booking_Measures table in the Data Pane.
  2. In the Home ribbon, click on New measure. 
  3. In the formula bar type the DAX code below, then click  to commit and  to close. oldCR = AVERAGE(hotel_bookings[is_canceled])
  4. In the Booking_Measures table, Right click on the ellipses (â€¦) next to column 1 and delete from model. Confirm deletion in the pop-out screen.
  5. Repeat the steps 1-3 for the remaining measures.
Note: Must follow the code order below to avoid Error messages.

oldLT = CALCULATE(AVERAGE(hotel_bookings[lead_time]))
 
CR = 
VAR CurrentLeadTime = [oldLT] 
VAR AdjustedLeadTime = CurrentLeadTime + SELECTEDVALUE('LeadTimeAdjustment'[LeadTimeAdjustment])*CurrentLeadTime
VAR CurrentCancelRate = [oldCR] 
VAR DecayFactor = 0.034 
VAR AdjustedCancelRate = CurrentCancelRate * EXP(DecayFactor * (AdjustedLeadTime - CurrentLeadTime))
RETURN
    IF(
        AdjustedLeadTime = CurrentLeadTime,
        CurrentCancelRate,  
        AdjustedCancelRate 
    )
 
LT = [oldLT] + SELECTEDVALUE('LeadTimeAdjustment'[LeadTimeAdjustment])*[oldLT] 
 
% Change = 
VAR AdjustedCancelRate = [CR]
VAR CancelRate = [oldCR]
RETURN
    IF(
        ISBLANK(CancelRate),
        BLANK(),
        DIVIDE(AdjustedCancelRate - CancelRate, CancelRate, 0)
    )
 
RSV = COUNTROWS(hotel_bookings)
 
ADR = AVERAGE('hotel_bookings'[adr_USD])

LT YoY = 
VAR LYLT =
CALCULATE([oldLT], SAMEPERIODLASTYEAR('Calendar'[Date]))
RETURN
IF(ISBLANK(LYLT),BLANK(),DIVIDE([oldLT]-LYLT,LYLT))
 
RSV YoY = 
VAR LYRSV =
CALCULATE([RSV], SAMEPERIODLASTYEAR('Calendar'[Date]))
RETURN
IF(ISBLANK(LYRSV),BLANK(),DIVIDE([RSV]-LYRSV,LYRSV))
 
ADR YoY = 
VAR LYADR =
CALCULATE([ADR], SAMEPERIODLASTYEAR('Calendar'[Date]))
RETURN
IF(ISBLANK(LYADR),BLANK(),(DIVIDE([ADR]-LYADR,LYADR)))
 
CR YoY = 
VAR LYCR =
CALCULATE([CR], SAMEPERIODLASTYEAR('Calendar'[Date]))
RETURN
IF(ISBLANK(LYCR),BLANK(),DIVIDE([CR]-LYCR,LYCR))
 
Arrow RSV = SWITCH(TRUE(), [RSV YoY] <0, UNICHAR(9662),[RSV YoY] =0, UNICHAR(9666), UNICHAR(9652))
 
Arrow LT = SWITCH(TRUE(), [LT YoY] <0, UNICHAR(9662),[LT YoY] =0, UNICHAR(9666), UNICHAR(9652))

Arrow % Change = SWITCH(TRUE(), [% Change] <0, UNICHAR(9662),[% Change] =0, UNICHAR(9666), UNICHAR(9652))
